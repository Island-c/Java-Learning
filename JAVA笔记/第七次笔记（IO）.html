ByteArray IO

任何对象都可以转成字节数组，存在内存而不是硬盘，不宜太大，IO不用关闭。

    public static void main(String[] args)
    {
        /*字节数组输入*/
        byte[] src="talk is cheap show me the code".getBytes(); //转字节数组
        InputStream is =null;
        try {
            is=new ByteArrayInputStream(src);
            
            byte[] flush=new byte[5];
            int len=-1;
            while((len=is.read(flush))!=-1) {
                String str=new String(flush);
                System.out.println(str);
            }
        }catch(IOException e)
        {
            e.printStackTrace();
        }finally {
            //也可以不关闭
            try {
                if(is!=null)
                    is.close();
            }catch(IOException e) {
                e.printStackTrace();
            }
        }
        
        /*字节数组输出*/
        byte[] dest = null;
        ByteArrayOutputStream baos = null;
        try {
            baos=new ByteArrayOutputStream(); //不需要关联源
            String msg="show me your code";
            byte[]datas = msg.getBytes();
            baos.write(datas,0,datas.length);
            baos.flush();
            dest=baos.toByteArray(); //从缓冲区获取数据
            System.out.println(dest.length + "->>>" + new String(dest,0,baos.size()));
            
        }catch(FileNotFoundException e) {
            e.printStackTrace();
        }catch(IOException e) {
            e.printStackTrace();
        }finally {
            //也可以不关闭
            try {
                if(baos!=null)
                    baos.close();
            }catch(IOException e) {
                e.printStackTrace();
            }
        }
        
    }

package cn.lc.IO;
/**
 * 模拟咖啡
 * 1.抽象组件：需要装饰的抽象对象（接口 或 抽象父类）
 * 2.具体组件：要装饰的对象
 * 3.抽象装饰类：包含了对抽象组件的引用 和 装饰者共有方法
 * 4.具体装饰类： 被装饰的对象
 * @author Trail
 *
 */
public class DecorateTest2 {
    public static void main(String[] args) {
        Drink coffee=new Coffee();
        
        Drink suger=new Suger(coffee);//装饰
        System.out.println(suger.info() + "->>" + suger.cost());
        
        Drink milk=new Milk(coffee);//装饰
        System.out.println(milk.info() + "->>" + milk .cost());
        
        milk = new Milk(suger);//装饰
        System.out.println(milk.info() + "->>" + milk .cost());
    }
}
//具体装饰类
class Suger extends Decorate{
    public Suger(Drink drink)
    {
         super(drink);
    }
    
    @Override
    public double cost() {
        return super.cost()*2;
    }

    @Override
    public String info() {
        return super.info()+"加入了蔗糖";
    }
    
}
class Milk extends Decorate{
    public Milk(Drink drink)
    {
         super(drink);
    }
    
    @Override
    public double cost() {
        return super.cost()*4;
    }

    @Override
    public String info() {
        return super.info()+"加入了牛奶";
    }
    
}


//抽象装饰类
abstract class Decorate implements Drink{
    private Drink drink;
    public Decorate(Drink drink)
    {
        this.drink = drink;
    }
    
    @Override
    public double cost() {
        return this.drink.cost();
    }

    @Override
    public String info() {
        return this.drink.info();
    }
    
}

//具体组件
class Coffee implements Drink{
    private String name= "原味咖啡";
    
    @Override
    public double cost() {
        return 10;
    }
    @Override
    public String info() {
        return name;
    }
}

interface Drink{
    double cost();
    String info();
    
}
